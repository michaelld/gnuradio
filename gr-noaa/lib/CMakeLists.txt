########################################################################
# Setup library
########################################################################
list(APPEND noaa_sources
    hrpt_decoder_impl.cc
    hrpt_deframer_impl.cc
    hrpt_pll_cf_impl.cc
)

#Add Windows DLL resource file if using MSVC
if(MSVC)
    include(${CMAKE_SOURCE_DIR}/cmake/Modules/GrVersion.cmake)

    configure_file(
        ${CMAKE_CURRENT_SOURCE_DIR}/gnuradio-noaa.rc.in
        ${CMAKE_CURRENT_BINARY_DIR}/gnuradio-noaa.rc
    @ONLY)

    list(APPEND gr_noaa_sources
        ${CMAKE_CURRENT_BINARY_DIR}/gnuradio-noaa.rc
    )
endif(MSVC)

GR_LIBRARY_FOO(gnuradio-noaa
    RUNTIME_COMPONENT "noaa_runtime"
    DEVEL_COMPONENT "noaa_devel"
    AUTO_DEPS_VAR ENABLE_GR_NOAA
    SOURCES ${noaa_sources}
)
